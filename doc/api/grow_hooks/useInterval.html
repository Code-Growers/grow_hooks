void useInterval(VoidCallback callback, Duration duration,
    {bool enabled = true}) {
  useEffect(() {
    if (enabled) {
      final timer = Timer.periodic(duration, (_) {
        callback();
      });
      return () => timer.cancel();
    }
    return null;
  }, [callback, enabled]);
}
